`; return; } const lines = numbersText.split('\n'); let count = 0; let totalCents = 0; for(let i = 0; i < lines.length; i++){ let clean = lines[i].trim().replace(/,/g,''); if(clean === '') continue; let num = parseFloat(clean); if(isNaN(num)){ inputSummaryBox.className = 'result-box result-error'; inputSummaryBox.innerHTML = `

`; return; } count++; totalCents += Math.round(num * 100); } let total = (totalCents / 100).toFixed(2); inputSummaryBox.className = 'result-box result-success'; inputSummaryBox.innerHTML = `
Input Summary

${count}
Total of All Numbers
